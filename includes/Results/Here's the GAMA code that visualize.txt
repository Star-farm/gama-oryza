// Modified plot species with separate aspects for each irrigation regime
species plot {
    int plot_id;
    string regime;
    float water_level_s1 <- 0.0;
    float water_level_s2 <- 0.0;
    
    // Aspect for CF plots
    aspect cf_display {
        if (regime = REGIME_CF) {
            rgb plot_color <- calculate_water_color_cf();
            draw shape color: plot_color border: #black width: 2;
            draw string(plot_id) color: #white size: 12 at: location;
            
            string scenario_1_name <- SCENARIO_NAMES[scenario_1];
            string scenario_2_name <- SCENARIO_NAMES[scenario_2];
            
            draw string(scenario_1_name + ": " + water_level_s1 with_precision 1 + "mm")
                color: #white size: 10 at: {location.x, location.y - 10};
            draw string(scenario_2_name + ": " + water_level_s2 with_precision 1 + "mm")
                color: #yellow size: 10 at: {location.x, location.y - 20};
            
            // Add regime label
            draw string("CF") color: #red size: 8 at: {location.x, location.y + 15};
        }
    }
    
    // Aspect for AWD plots  
    aspect awd_display {
        if (regime = REGIME_AWD) {
            rgb plot_color <- calculate_water_color_awd();
            draw shape color: plot_color border: #black width: 2;
            draw string(plot_id) color: #white size: 12 at: location;
            
            string scenario_1_name <- SCENARIO_NAMES[scenario_1];
            string scenario_2_name <- SCENARIO_NAMES[scenario_2];
            
            draw string(scenario_1_name + ": " + water_level_s1 with_precision 1 + "mm")
                color: #white size: 10 at: {location.x, location.y - 10};
            draw string(scenario_2_name + ": " + water_level_s2 with_precision 1 + "mm")
                color: #yellow size: 10 at: {location.x, location.y - 20};
                
            // Add regime label
            draw string("AWD") color: #green size: 8 at: {location.x, location.y + 15};
        }
    }
    
    // Color calculation based on scenario 1 water level for CF
    rgb calculate_water_color_cf {
        float water_level <- water_level_s1; // Use scenario 1 as base for color
        if (water_level <= 0.0) {
            return rgb(139, 119, 101); // Brown for dry
        } else {
            float water_ratio <- water_level / max_water_level;
            int blue_intensity <- int(255 - (water_ratio * 205));
            blue_intensity <- max([50, min([255, blue_intensity])]);
            int green_component <- int(blue_intensity * 0.3);
            return rgb(0, green_component, blue_intensity);
        }
    }
    
    // Color calculation based on scenario 1 water level for AWD  
    rgb calculate_water_color_awd {
        float water_level <- water_level_s1; // Use scenario 1 as base for color
        if (water_level <= 0.0) {
            return rgb(139, 119, 101); // Brown for dry
        } else {
            float water_ratio <- water_level / max_water_level;
            int blue_intensity <- int(255 - (water_ratio * 205));
            blue_intensity <- max([50, min([255, blue_intensity])]);
            int green_component <- int(blue_intensity * 0.4); // Slightly different green tone for AWD
            return rgb(10, green_component, blue_intensity); // Slightly different base for distinction
        }
    }
}
